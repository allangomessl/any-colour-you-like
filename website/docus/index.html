UNIVERSE UI

Install

# npm
npm i @universe-ui/core styled-components
# yarn
yarn add @universe-ui/core styled-components

Setup

in your app.tsx

import { Root } from "@universe-ui/core"
import { darkSide } from "@universe-ui/themes"
export const App = () => {
  <Root theme={darkSide}>
    {...}
  </Root>
}

Using Router

import { Link } from "@universe-ui/core"
import { Link as RouterLink } from 'react-router-dom'
Link.initialize(RouterLink)
// and after
<Link to="path" /> // render as Router Link
<Button to="path" /> // render as Router Link
<Button href="path" /> // render as normal anchor
