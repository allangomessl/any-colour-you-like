# Integration

<br /><br />

```tsx live
() => {
  const form = useRef()
  const [submitData, setSubmitData] = useState('')
  const handleSubmit = useCallback((data) => {
    setSubmitData(JSON.stringify(data))
    return data
  }, [setSubmitData])
  return (
    <Form ref={form} onSubmit={handleSubmit} onSuccess={console.log}>
      <Field.Text label='Name' name='name' />
      <Field.Select label='Sex' name='sex' items={['M', 'F']} />
      <Field.Date label='Birth Date' name='birthDate' />
      <Form path='address'>
        <Field.Text label='Street' name='street' />
      </Form>
      <Form path='phones[0]'>
        <Field.Masked type='phone' label='Phone 1' name='number' />
      </Form>
      <Row>
        <Button onClick={form?.submit}>Save</Button>
      </Row>
      <Row margin='T4'>
        <code>
          {submitData}
        </code>
      </Row>
    </Form>
  )
}
```
