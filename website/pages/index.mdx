## Install

```sh
# npm
npm i paintbox styled-components

# yarn
yarn add paintbox styled-components
```


### Setup

in your `app.tsx`
```tsx
import { Root } from "paintbox"
import { darkSide } from "@paintbox/themes"

export const App = () => {
  <Root theme={darkSide}>
    {...}
  </Root>
}
```


### Using Router

```tsx
import { Link } from "paintbox"
import { Link as RouterLink } from 'react-router-dom'

Link.initialize(RouterLink)

// and after

<Link to="path" /> // render as Router Link
<Button to="path" /> // render as Router Link
<Button href="path" /> // render as normal anchor
```
